models.py Przykładowy model
from django.db import models from django.contrib.auth.models import User class Service(models.Model): SERVICE_TYPES = [ ('OUT', 'Outsourcing IT'), ('DEV', 'Tworzenie oprogramowania'), ('ERP', 'Integracja ERP/CRM/BI'), # ... inne usługi ] title = models.CharField(max_length=100) service_type = models.CharField(max_length=3, choices=SERVICE_TYPES) description = models.TextField() icon = models.CharField(max_length=30) # Nazwa ikony z FontAwesome is_featured = models.BooleanField(default=False) created_at = models.DateTimeField(auto_now_add=True) def __str__(self): return self.title

views.py Przykładowy widok
from django.views.generic import ListView, DetailView from django.views.generic.edit import FormView from .models import Service, BlogPost from .forms import ContactForm class HomeView(ListView): model = Service template_name = 'home.html' context_object_name = 'services' def get_queryset(self): return Service.objects.filter(is_featured=True) def get_context_data(self, **kwargs): context = super().get_context_data(**kwargs) context['featured_posts'] = BlogPost.objects.filter(is_featured=True)[:3] return context class ServiceDetailView(DetailView): model = Service template_name = 'services/detail.html' context_object_name = 'service'

// Animacja scrollowania document.querySelectorAll('a[href^="#"]').forEach(anchor => { anchor.addEventListener('click', function(e) { e.preventDefault(); document.querySelector(this.getAttribute('href')).scrollIntoView({ behavior: 'smooth' }); }); }); // Obsługa formularza kontaktowego const contactForm = document.getElementById('contact-form'); if (contactForm) { contactForm.addEventListener('submit', async (e) => { e.preventDefault(); const formData = new FormData(contactForm); try { const response = await fetch('/api/contact/', { method: 'POST', body: formData }); if (response.ok) { showNotification('Wiadomość wysłana pomyślnie!', 'success'); contactForm.reset(); } else { throw new Error('Błąd podczas wysyłania'); } } catch (error) { showNotification('Wystąpił błąd. Spróbuj ponownie.', 'error'); console.error(error); } }); } // Efekt paralaksy window.addEventListener('scroll', () => { const scrollPosition = window.pageYOffset; const parallaxElements = document.querySelectorAll('[data-parallax]'); parallaxElements.forEach(element => { const speed = parseFloat(element.dataset.parallax); element.style.transform = `translateY(${scrollPosition * speed}px)`; }); });

Przykładowa animacja GSAP
gsap.registerPlugin(ScrollTrigger); // Animacja sekcji usług gsap.utils.toArray('.service-card').forEach((card, i) => { gsap.from(card, { scrollTrigger: { trigger: card, start: "top 80%", toggleActions: "play none none none" }, opacity: 0, y: 50, duration: 0.8, delay: i * 0.1 }); }); // Animacja logo const logo = document.querySelector('.logo'); gsap.to(logo, { duration: 2, rotate: 360, repeat: -1, ease: "power1.inOut", yoyo: true });
